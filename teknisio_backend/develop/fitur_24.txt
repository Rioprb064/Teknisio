#!/usr/bin/env bash
set -euo pipefail

BASE_URL="http://localhost:8080"

if [[ ! -f ".env" ]]; then
  echo ".env not found. Run this script from backend root folder."
  exit 1
fi

set -a
source .env
set +a

DB_HOST="${POSTGRES_HOST:-localhost}"
DB_PORT="${POSTGRES_PORT:-5433}"
DB_NAME="${POSTGRES_DB:-teknisio_db}"
DB_USER="${POSTGRES_USER:-teknisio_user}"
DB_PASSWORD="${POSTGRES_PASSWORD:-}"

if [[ -z "$DB_PASSWORD" ]]; then
  echo "POSTGRES_PASSWORD is empty. Check your .env file."
  exit 1
fi

TS=$(date +%s)

CUSTOMER_EMAIL="customer${TS}@example.com"
CUSTOMER_PHONE="0812${TS: -8}"

TECHNICIAN_EMAIL="technician${TS}@example.com"
TECHNICIAN_PHONE="0813${TS: -8}"

PASSWORD="Password123"

echo "Using test data:"
echo "Customer email   : $CUSTOMER_EMAIL"
echo "Technician email : $TECHNICIAN_EMAIL"
echo "Database         : $DB_HOST:$DB_PORT/$DB_NAME"
echo

echo "1. Register customer"
CUSTOMER_REGISTER_RESPONSE=$(curl -s -X POST "$BASE_URL/api/auth/register/customer" \
  -H "Content-Type: application/json" \
  -d "{
    \"name\": \"Customer BE24 Test\",
    \"email\": \"$CUSTOMER_EMAIL\",
    \"phoneNumber\": \"$CUSTOMER_PHONE\",
    \"password\": \"$PASSWORD\",
    \"address\": \"Jl. Customer BE24 No. 1\"
  }")

echo "$CUSTOMER_REGISTER_RESPONSE" | jq

if [[ "$(echo "$CUSTOMER_REGISTER_RESPONSE" | jq -r '.success')" != "true" ]]; then
  echo "Customer registration failed"
  exit 1
fi

echo

echo "2. Register technician"
TECHNICIAN_REGISTER_RESPONSE=$(curl -s -X POST "$BASE_URL/api/auth/register/technician" \
  -H "Content-Type: application/json" \
  -d "{
    \"name\": \"Technician BE24 Test\",
    \"email\": \"$TECHNICIAN_EMAIL\",
    \"phoneNumber\": \"$TECHNICIAN_PHONE\",
    \"password\": \"$PASSWORD\",
    \"address\": \"Jl. Technician BE24 No. 2\",
    \"description\": \"Air Conditioner and Refrigerator technician\"
  }")

echo "$TECHNICIAN_REGISTER_RESPONSE" | jq

if [[ "$(echo "$TECHNICIAN_REGISTER_RESPONSE" | jq -r '.success')" != "true" ]]; then
  echo "Technician registration failed"
  exit 1
fi

TECHNICIAN_PROFILE_ID=$(echo "$TECHNICIAN_REGISTER_RESPONSE" | jq -r '.data.technicianProfileId')

if [[ "$TECHNICIAN_PROFILE_ID" == "null" || -z "$TECHNICIAN_PROFILE_ID" ]]; then
  echo "technicianProfileId not found"
  exit 1
fi

echo "Technician profile ID: $TECHNICIAN_PROFILE_ID"
echo

echo "3. Login customer and auto-save token"
CUSTOMER_LOGIN_RESPONSE=$(curl -s -X POST "$BASE_URL/api/auth/login" \
  -H "Content-Type: application/json" \
  -d "{
    \"email\": \"$CUSTOMER_EMAIL\",
    \"password\": \"$PASSWORD\"
  }")

CUSTOMER_TOKEN=$(echo "$CUSTOMER_LOGIN_RESPONSE" | jq -r '.data.accessToken')

echo "$CUSTOMER_LOGIN_RESPONSE" | jq '.data.accessToken = "<hidden>" | .data.refreshToken = "<hidden>"'

if [[ "$CUSTOMER_TOKEN" == "null" || -z "$CUSTOMER_TOKEN" ]]; then
  echo "Customer token not found"
  exit 1
fi

echo

echo "4. Login technician and auto-save token"
TECHNICIAN_LOGIN_RESPONSE=$(curl -s -X POST "$BASE_URL/api/auth/login" \
  -H "Content-Type: application/json" \
  -d "{
    \"email\": \"$TECHNICIAN_EMAIL\",
    \"password\": \"$PASSWORD\"
  }")

TECHNICIAN_TOKEN=$(echo "$TECHNICIAN_LOGIN_RESPONSE" | jq -r '.data.accessToken')

echo "$TECHNICIAN_LOGIN_RESPONSE" | jq '.data.accessToken = "<hidden>" | .data.refreshToken = "<hidden>"'

if [[ "$TECHNICIAN_TOKEN" == "null" || -z "$TECHNICIAN_TOKEN" ]]; then
  echo "Technician token not found"
  exit 1
fi

echo

echo "5. Get device categories"
DEVICE_CATEGORIES_RESPONSE=$(curl -s -X GET "$BASE_URL/api/device-categories")
echo "$DEVICE_CATEGORIES_RESPONSE" | jq

AIR_CONDITIONER_ID=$(echo "$DEVICE_CATEGORIES_RESPONSE" | jq -r '.data[] | select(.name == "Air Conditioner") | .deviceCategoryId')
REFRIGERATOR_ID=$(echo "$DEVICE_CATEGORIES_RESPONSE" | jq -r '.data[] | select(.name == "Refrigerator") | .deviceCategoryId')

if [[ "$AIR_CONDITIONER_ID" == "null" || -z "$AIR_CONDITIONER_ID" ]]; then
  echo "Air Conditioner category not found"
  exit 1
fi

if [[ "$REFRIGERATOR_ID" == "null" || -z "$REFRIGERATOR_ID" ]]; then
  echo "Refrigerator category not found"
  exit 1
fi

echo "Air Conditioner ID: $AIR_CONDITIONER_ID"
echo "Refrigerator ID   : $REFRIGERATOR_ID"
echo

echo "6. Auto insert technician skills into real backend database"
PGPASSWORD="$DB_PASSWORD" psql \
  -h "$DB_HOST" \
  -p "$DB_PORT" \
  -U "$DB_USER" \
  -d "$DB_NAME" \
  -v ON_ERROR_STOP=1 \
  -c "
INSERT INTO teknisi_kategori_layanan (
  id_teknisi_profile,
  id_kategori,
  aktif
)
VALUES
  ('$TECHNICIAN_PROFILE_ID', '$AIR_CONDITIONER_ID', true),
  ('$TECHNICIAN_PROFILE_ID', '$REFRIGERATOR_ID', true)
ON CONFLICT (id_teknisi_profile, id_kategori)
DO UPDATE SET
  aktif = true,
  updated_at = CURRENT_TIMESTAMP;
"

echo

echo "7. Verify inserted technician skills"
PGPASSWORD="$DB_PASSWORD" psql \
  -h "$DB_HOST" \
  -p "$DB_PORT" \
  -U "$DB_USER" \
  -d "$DB_NAME" \
  -c "
SELECT
  u.nama AS technician_name,
  k.nama_kategori AS device_category,
  tkl.aktif
FROM teknisi_kategori_layanan tkl
JOIN teknisi_profile tp
  ON tp.id_teknisi_profile = tkl.id_teknisi_profile
JOIN users u
  ON u.id_user = tp.id_user
JOIN kategori_layanan k
  ON k.id_kategori = tkl.id_kategori
WHERE tkl.id_teknisi_profile = '$TECHNICIAN_PROFILE_ID'
ORDER BY k.nama_kategori;
"

echo

echo "8. Test BE-24 without token, expected Unauthorized"
curl -s -X GET "$BASE_URL/api/customers/technicians?deviceCategoryId=$AIR_CONDITIONER_ID" | jq
echo

echo "9. Test BE-24 with technician token, expected Forbidden"
curl -s -X GET "$BASE_URL/api/customers/technicians?deviceCategoryId=$AIR_CONDITIONER_ID" \
  -H "Authorization: Bearer $TECHNICIAN_TOKEN" | jq
echo

echo "10. Test BE-24 with customer token, expected success and technician appears"
BE24_RESPONSE=$(curl -s -X GET "$BASE_URL/api/customers/technicians?deviceCategoryId=$AIR_CONDITIONER_ID" \
  -H "Authorization: Bearer $CUSTOMER_TOKEN")

echo "$BE24_RESPONSE" | jq

TECHNICIAN_COUNT=$(echo "$BE24_RESPONSE" | jq '.data | length')

echo

if [[ "$TECHNICIAN_COUNT" -gt 0 ]]; then
  echo "BE-24 SUCCESS: technician appears"
else
  echo "BE-24 WARNING: response success, but technician data is empty"
  echo "Check CustomerTechnicianService or TeknisiKategoriLayananRepository"
fi

echo

echo "11. Test invalid category with customer token, expected Device category not found"
curl -s -X GET "$BASE_URL/api/customers/technicians?deviceCategoryId=00000000-0000-0000-0000-000000000000" \
  -H "Authorization: Bearer $CUSTOMER_TOKEN" | jq

echo
echo "Done."
