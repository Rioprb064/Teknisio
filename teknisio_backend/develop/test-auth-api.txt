#!/usr/bin/env bash
set -Eeuo pipefail

BASE_URL="${BASE_URL:-http://localhost:8080}"

PASS_COUNT=0
FAIL_COUNT=0
LAST_STATUS=""
LAST_BODY=""

require_command() {
  if ! command -v "$1" >/dev/null 2>&1; then
    echo "Missing command: $1"
    exit 1
  fi
}

require_command curl
require_command python3

redact_json() {
  python3 -c '
import json
import sys

raw = sys.stdin.read()

try:
    if not raw.strip():
        print("")
        sys.exit(0)

    data = json.loads(raw)

    def scrub(value):
        if isinstance(value, dict):
            return {
                key: "<hidden>" if key in ("accessToken", "refreshToken") else scrub(val)
                for key, val in value.items()
            }
        if isinstance(value, list):
            return [scrub(item) for item in value]
        return value

    print(json.dumps(scrub(data), indent=2))
except Exception:
    print(raw)
'
}

json_get() {
  local key="$1"

  python3 -c '
import json
import sys

key = sys.argv[1]
raw = sys.stdin.read()

try:
    data = json.loads(raw)
    value = ""

    if isinstance(data, dict):
        value = data.get(key, "")

        if value == "" and isinstance(data.get("data"), dict):
            value = data["data"].get(key, "")

    if isinstance(value, bool):
        print(str(value).lower())
    elif value is None:
        print("")
    else:
        print(value)
except Exception:
    print("")
' "$key"
}

call_api() {
  local name="$1"
  local method="$2"
  local path="$3"
  local expected="$4"
  local token="${5:-}"
  local data="${6:-}"

  echo
  echo "============================================================"
  echo "TEST: $name"
  echo "$method $path"
  echo "Expected HTTP: $expected"
  echo "============================================================"

  local body_file
  body_file="$(mktemp)"

  local args=(
    -sS
    -X "$method"
    "$BASE_URL$path"
    -H "Content-Type: application/json"
    -o "$body_file"
    -w "%{http_code}"
  )

  if [[ -n "$token" ]]; then
    args+=(-H "Authorization: Bearer $token")
  fi

  if [[ -n "$data" ]]; then
    args+=(-d "$data")
  fi

  LAST_STATUS="$(curl "${args[@]}")"
  LAST_BODY="$(cat "$body_file")"
  rm -f "$body_file"

  local expected_success="false"

  if [[ "$LAST_STATUS" =~ ^2 ]]; then
    expected_success="true"
  fi

  local actual_success
  actual_success="$(printf '%s' "$LAST_BODY" | json_get success)"

  local http_ok="false"
  local wrapper_ok="false"

  if [[ "$LAST_STATUS" =~ ^($expected)$ ]]; then
    http_ok="true"
  fi

  if [[ "$actual_success" == "$expected_success" ]]; then
    wrapper_ok="true"
  fi

  if [[ "$http_ok" == "true" && "$wrapper_ok" == "true" ]]; then
    echo "PASS - HTTP $LAST_STATUS, success=$actual_success"
    PASS_COUNT=$((PASS_COUNT + 1))
  else
    echo "FAIL - HTTP $LAST_STATUS, success=$actual_success"
    echo "Expected success=$expected_success"
    FAIL_COUNT=$((FAIL_COUNT + 1))
  fi

  echo
  echo "Response:"
  printf '%s' "$LAST_BODY" | redact_json
}

echo "Checking server: $BASE_URL"

if ! curl -sS "$BASE_URL/api/auth/profile" >/dev/null 2>&1; then
  echo "Server is not reachable. Run the app first:"
  echo "./gradlew bootRun"
  exit 1
fi

RUN_ID="$(date +%s)"
SUFFIX="${RUN_ID: -7}"

CUSTOMER_EMAIL="customer${RUN_ID}@example.com"
CUSTOMER_PHONE="0812${SUFFIX}"
TECHNICIAN_EMAIL="technician${RUN_ID}@example.com"
TECHNICIAN_PHONE="0813${SUFFIX}"
PASSWORD="password123"

CUSTOMER_ACCESS_TOKEN=""
CUSTOMER_REFRESH_TOKEN=""
TECHNICIAN_ACCESS_TOKEN=""
TECHNICIAN_REFRESH_TOKEN=""

echo
echo "Using test data:"
echo "Customer email   : $CUSTOMER_EMAIL"
echo "Customer phone   : $CUSTOMER_PHONE"
echo "Technician email : $TECHNICIAN_EMAIL"
echo "Technician phone : $TECHNICIAN_PHONE"

call_api \
  "Profile without token should be unauthorized" \
  "GET" \
  "/api/auth/profile" \
  "401"

call_api \
  "Register customer validation error" \
  "POST" \
  "/api/auth/register/customer" \
  "400" \
  "" \
  '{}'

call_api \
  "Register customer success" \
  "POST" \
  "/api/auth/register/customer" \
  "201" \
  "" \
  "{
    \"name\": \"Customer Test\",
    \"email\": \"$CUSTOMER_EMAIL\",
    \"phoneNumber\": \"$CUSTOMER_PHONE\",
    \"password\": \"$PASSWORD\",
    \"address\": \"Jl. Customer Test No. 1\"
  }"

call_api \
  "Register customer duplicate email/phone should fail" \
  "POST" \
  "/api/auth/register/customer" \
  "400" \
  "" \
  "{
    \"name\": \"Customer Duplicate\",
    \"email\": \"$CUSTOMER_EMAIL\",
    \"phoneNumber\": \"$CUSTOMER_PHONE\",
    \"password\": \"$PASSWORD\",
    \"address\": \"Jl. Customer Test No. 1\"
  }"

call_api \
  "Login customer with wrong password should fail" \
  "POST" \
  "/api/auth/login" \
  "401" \
  "" \
  "{
    \"email\": \"$CUSTOMER_EMAIL\",
    \"password\": \"wrongpassword\"
  }"

call_api \
  "Login customer success" \
  "POST" \
  "/api/auth/login" \
  "200" \
  "" \
  "{
    \"email\": \"$CUSTOMER_EMAIL\",
    \"password\": \"$PASSWORD\"
  }"

CUSTOMER_ACCESS_TOKEN="$(printf '%s' "$LAST_BODY" | json_get accessToken)"
CUSTOMER_REFRESH_TOKEN="$(printf '%s' "$LAST_BODY" | json_get refreshToken)"

if [[ -z "$CUSTOMER_ACCESS_TOKEN" || -z "$CUSTOMER_REFRESH_TOKEN" ]]; then
  echo
  echo "Failed to extract customer tokens."
  echo "Raw login response:"
  printf '%s' "$LAST_BODY" | redact_json
  exit 1
fi

call_api \
  "Get customer profile success" \
  "GET" \
  "/api/auth/profile" \
  "200" \
  "$CUSTOMER_ACCESS_TOKEN"

call_api \
  "Refresh customer access token success" \
  "POST" \
  "/api/auth/refresh" \
  "200" \
  "" \
  "{
    \"refreshToken\": \"$CUSTOMER_REFRESH_TOKEN\"
  }"

call_api \
  "Customer cannot access technician protected route" \
  "GET" \
  "/api/technicians/protected-check" \
  "403" \
  "$CUSTOMER_ACCESS_TOKEN"

call_api \
  "Logout customer success" \
  "POST" \
  "/api/auth/logout" \
  "200" \
  "$CUSTOMER_ACCESS_TOKEN" \
  "{
    \"refreshToken\": \"$CUSTOMER_REFRESH_TOKEN\"
  }"

call_api \
  "Refresh after logout should fail" \
  "POST" \
  "/api/auth/refresh" \
  "401" \
  "" \
  "{
    \"refreshToken\": \"$CUSTOMER_REFRESH_TOKEN\"
  }"

call_api \
  "Register technician success" \
  "POST" \
  "/api/auth/register/technician" \
  "201" \
  "" \
  "{
    \"name\": \"Technician Test\",
    \"email\": \"$TECHNICIAN_EMAIL\",
    \"phoneNumber\": \"$TECHNICIAN_PHONE\",
    \"password\": \"$PASSWORD\",
    \"address\": \"Jl. Technician Test No. 2\",
    \"description\": \"AC and refrigerator technician\"
  }"

call_api \
  "Login technician success" \
  "POST" \
  "/api/auth/login" \
  "200" \
  "" \
  "{
    \"email\": \"$TECHNICIAN_EMAIL\",
    \"password\": \"$PASSWORD\"
  }"

TECHNICIAN_ACCESS_TOKEN="$(printf '%s' "$LAST_BODY" | json_get accessToken)"
TECHNICIAN_REFRESH_TOKEN="$(printf '%s' "$LAST_BODY" | json_get refreshToken)"

if [[ -z "$TECHNICIAN_ACCESS_TOKEN" || -z "$TECHNICIAN_REFRESH_TOKEN" ]]; then
  echo
  echo "Failed to extract technician tokens."
  echo "Raw login response:"
  printf '%s' "$LAST_BODY" | redact_json
  exit 1
fi

call_api \
  "Get technician profile success" \
  "GET" \
  "/api/auth/profile" \
  "200" \
  "$TECHNICIAN_ACCESS_TOKEN"

call_api \
  "Technician cannot access customer protected route" \
  "GET" \
  "/api/customers/protected-check" \
  "403" \
  "$TECHNICIAN_ACCESS_TOKEN"

call_api \
  "Logout technician success" \
  "POST" \
  "/api/auth/logout" \
  "200" \
  "$TECHNICIAN_ACCESS_TOKEN" \
  "{
    \"refreshToken\": \"$TECHNICIAN_REFRESH_TOKEN\"
  }"

echo
echo "============================================================"
echo "SUMMARY"
echo "============================================================"
echo "Passed: $PASS_COUNT"
echo "Failed: $FAIL_COUNT"

if [[ "$FAIL_COUNT" -gt 0 ]]; then
  exit 1
fi

echo "All auth API tests passed."
