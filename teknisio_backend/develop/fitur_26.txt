#!/usr/bin/env bash
set -euo pipefail

BASE_URL="http://localhost:8080"

SCRIPT_DIR="$(cd "$(dirname "${BASH_SOURCE[0]}")" && pwd)"

if [[ -f ".env" ]]; then
  ENV_FILE=".env"
elif [[ -f "$SCRIPT_DIR/../.env" ]]; then
  ENV_FILE="$SCRIPT_DIR/../.env"
else
  echo ".env not found. Run this script from backend root folder."
  exit 1
fi

for cmd in curl jq psql; do
  if ! command -v "$cmd" >/dev/null 2>&1; then
    echo "$cmd is not installed"
    exit 1
  fi
done

set -a
source "$ENV_FILE"
set +a

DB_HOST="${POSTGRES_HOST:-localhost}"
DB_PORT="${POSTGRES_PORT:-5433}"
DB_NAME="${POSTGRES_DB:-teknisio_db}"
DB_USER="${POSTGRES_USER:-teknisio_user}"
DB_PASSWORD="${POSTGRES_PASSWORD:-}"

if [[ -z "$DB_PASSWORD" ]]; then
  echo "POSTGRES_PASSWORD is empty. Check your .env file."
  exit 1
fi

TS=$(date +%s)

CUSTOMER_EMAIL="customerbe26${TS}@example.com"
CUSTOMER_PHONE="0812${TS: -8}"
PASSWORD="Password123"

echo "Using test data:"
echo "Customer email : $CUSTOMER_EMAIL"
echo "Database       : $DB_HOST:$DB_PORT/$DB_NAME"
echo

echo "1. Register customer"
CUSTOMER_REGISTER_RESPONSE=$(curl -s -X POST "$BASE_URL/api/auth/register/customer" \
  -H "Content-Type: application/json" \
  -d "{
    \"name\": \"Customer BE26 Test\",
    \"email\": \"$CUSTOMER_EMAIL\",
    \"phoneNumber\": \"$CUSTOMER_PHONE\",
    \"password\": \"$PASSWORD\",
    \"address\": \"Jl. Customer BE26 No. 1\"
  }")

echo "$CUSTOMER_REGISTER_RESPONSE" | jq

if [[ "$(echo "$CUSTOMER_REGISTER_RESPONSE" | jq -r '.success')" != "true" ]]; then
  echo "Customer registration failed"
  exit 1
fi

echo

echo "2. Login customer"
CUSTOMER_LOGIN_RESPONSE=$(curl -s -X POST "$BASE_URL/api/auth/login" \
  -H "Content-Type: application/json" \
  -d "{
    \"email\": \"$CUSTOMER_EMAIL\",
    \"password\": \"$PASSWORD\"
  }")

CUSTOMER_TOKEN=$(echo "$CUSTOMER_LOGIN_RESPONSE" | jq -r '.data.accessToken')
echo "$CUSTOMER_LOGIN_RESPONSE" | jq '.data.accessToken = "<hidden>" | .data.refreshToken = "<hidden>"'

if [[ "$CUSTOMER_TOKEN" == "null" || -z "$CUSTOMER_TOKEN" ]]; then
  echo "Customer token not found"
  exit 1
fi

echo

register_technician() {
  local name="$1"
  local email="$2"
  local phone="$3"
  local description="$4"

  local response
  response=$(curl -s -X POST "$BASE_URL/api/auth/register/technician" \
    -H "Content-Type: application/json" \
    -d "{
      \"name\": \"$name\",
      \"email\": \"$email\",
      \"phoneNumber\": \"$phone\",
      \"password\": \"$PASSWORD\",
      \"address\": \"Jl. $name No. 1\",
      \"description\": \"$description\"
    }")

  echo "$response" | jq >&2

  if [[ "$(echo "$response" | jq -r '.success')" != "true" ]]; then
    echo "Technician registration failed for $name" >&2
    exit 1
  fi

  echo "$response" | jq -r '.data.technicianProfileId'
}

echo "3. Register 3 technicians"

TECH_A_EMAIL="techbe26a${TS}@example.com"
TECH_B_EMAIL="techbe26b${TS}@example.com"
TECH_C_EMAIL="techbe26c${TS}@example.com"

TECH_A_PHONE="0821${TS: -8}"
TECH_B_PHONE="0822${TS: -8}"
TECH_C_PHONE="0823${TS: -8}"

echo "3.1 Register Technician BE26 Alpha"
TECH_A_ID=$(register_technician \
  "Technician BE26 Alpha" \
  "$TECH_A_EMAIL" \
  "$TECH_A_PHONE" \
  "BE26 Alpha - high rating medium jobs")

echo "Technician A ID: $TECH_A_ID"
echo

echo "3.2 Register Technician BE26 Bravo"
TECH_B_ID=$(register_technician \
  "Technician BE26 Bravo" \
  "$TECH_B_EMAIL" \
  "$TECH_B_PHONE" \
  "BE26 Bravo - medium rating high jobs")

echo "Technician B ID: $TECH_B_ID"
echo

echo "3.3 Register Technician BE26 Charlie"
TECH_C_ID=$(register_technician \
  "Technician BE26 Charlie" \
  "$TECH_C_EMAIL" \
  "$TECH_C_PHONE" \
  "BE26 Charlie - highest rating low jobs offline")

echo "Technician C ID: $TECH_C_ID"
echo

echo "4. Login technician A for Forbidden test"
TECH_A_LOGIN_RESPONSE=$(curl -s -X POST "$BASE_URL/api/auth/login" \
  -H "Content-Type: application/json" \
  -d "{
    \"email\": \"$TECH_A_EMAIL\",
    \"password\": \"$PASSWORD\"
  }")

TECHNICIAN_TOKEN=$(echo "$TECH_A_LOGIN_RESPONSE" | jq -r '.data.accessToken')
echo "$TECH_A_LOGIN_RESPONSE" | jq '.data.accessToken = "<hidden>" | .data.refreshToken = "<hidden>"'

if [[ "$TECHNICIAN_TOKEN" == "null" || -z "$TECHNICIAN_TOKEN" ]]; then
  echo "Technician token not found"
  exit 1
fi

echo

echo "5. Get device categories"
DEVICE_CATEGORIES_RESPONSE=$(curl -s -X GET "$BASE_URL/api/device-categories")
echo "$DEVICE_CATEGORIES_RESPONSE" | jq

AIR_CONDITIONER_ID=$(echo "$DEVICE_CATEGORIES_RESPONSE" | jq -r '.data[] | select(.name == "Air Conditioner") | .deviceCategoryId')
REFRIGERATOR_ID=$(echo "$DEVICE_CATEGORIES_RESPONSE" | jq -r '.data[] | select(.name == "Refrigerator") | .deviceCategoryId')

if [[ "$AIR_CONDITIONER_ID" == "null" || -z "$AIR_CONDITIONER_ID" ]]; then
  echo "Air Conditioner category not found"
  exit 1
fi

if [[ "$REFRIGERATOR_ID" == "null" || -z "$REFRIGERATOR_ID" ]]; then
  echo "Refrigerator category not found"
  exit 1
fi

echo "Air Conditioner ID: $AIR_CONDITIONER_ID"
echo "Refrigerator ID   : $REFRIGERATOR_ID"
echo

echo "6. Set technician test data in database"
PGPASSWORD="$DB_PASSWORD" psql \
  -h "$DB_HOST" \
  -p "$DB_PORT" \
  -U "$DB_USER" \
  -d "$DB_NAME" \
  -v ON_ERROR_STOP=1 \
  -c "
UPDATE teknisi_profile
SET
  status_ketersediaan = 'ONLINE',
  rating_avg = 4.80,
  rating_count = 10,
  total_pekerjaan = 20
WHERE id_teknisi_profile = '$TECH_A_ID';

UPDATE teknisi_profile
SET
  status_ketersediaan = 'ONLINE',
  rating_avg = 4.50,
  rating_count = 100,
  total_pekerjaan = 50
WHERE id_teknisi_profile = '$TECH_B_ID';

UPDATE teknisi_profile
SET
  status_ketersediaan = 'OFFLINE',
  rating_avg = 5.00,
  rating_count = 2,
  total_pekerjaan = 5
WHERE id_teknisi_profile = '$TECH_C_ID';

INSERT INTO teknisi_kategori_layanan (
  id_teknisi_profile,
  id_kategori,
  aktif
)
VALUES
  ('$TECH_A_ID', '$AIR_CONDITIONER_ID', true),
  ('$TECH_A_ID', '$REFRIGERATOR_ID', true),
  ('$TECH_B_ID', '$AIR_CONDITIONER_ID', true),
  ('$TECH_C_ID', '$AIR_CONDITIONER_ID', true)
ON CONFLICT (id_teknisi_profile, id_kategori)
DO UPDATE SET
  aktif = true,
  updated_at = CURRENT_TIMESTAMP;
"

echo

echo "7. Verify technician data"
PGPASSWORD="$DB_PASSWORD" psql \
  -h "$DB_HOST" \
  -p "$DB_PORT" \
  -U "$DB_USER" \
  -d "$DB_NAME" \
  -c "
SELECT
  tp.id_teknisi_profile,
  u.nama,
  tp.status_ketersediaan,
  tp.rating_avg,
  tp.rating_count,
  tp.total_pekerjaan,
  k.nama_kategori
FROM teknisi_profile tp
JOIN users u
  ON u.id_user = tp.id_user
JOIN teknisi_kategori_layanan tkl
  ON tkl.id_teknisi_profile = tp.id_teknisi_profile
JOIN kategori_layanan k
  ON k.id_kategori = tkl.id_kategori
WHERE tp.id_teknisi_profile IN ('$TECH_A_ID', '$TECH_B_ID', '$TECH_C_ID')
ORDER BY u.nama, k.nama_kategori;
"

echo

echo "8. Test BE-26 without token, expected Unauthorized"
NO_TOKEN_RESPONSE=$(curl -s -X GET "$BASE_URL/api/customers/technicians?deviceCategoryId=$AIR_CONDITIONER_ID&availabilityStatus=ONLINE&sort=rating")
echo "$NO_TOKEN_RESPONSE" | jq

if [[ "$(echo "$NO_TOKEN_RESPONSE" | jq -r '.success')" != "false" ]]; then
  echo "No-token test failed: success should be false"
  exit 1
fi

if [[ "$(echo "$NO_TOKEN_RESPONSE" | jq -r '.message')" != "Unauthorized" ]]; then
  echo "No-token test failed: message should be Unauthorized"
  exit 1
fi

echo "No-token test SUCCESS"
echo

echo "9. Test BE-26 with technician token, expected Forbidden"
TECH_TOKEN_RESPONSE=$(curl -s -X GET "$BASE_URL/api/customers/technicians?deviceCategoryId=$AIR_CONDITIONER_ID&availabilityStatus=ONLINE&sort=rating" \
  -H "Authorization: Bearer $TECHNICIAN_TOKEN")

echo "$TECH_TOKEN_RESPONSE" | jq

if [[ "$(echo "$TECH_TOKEN_RESPONSE" | jq -r '.success')" != "false" ]]; then
  echo "Technician-token test failed: success should be false"
  exit 1
fi

if [[ "$(echo "$TECH_TOKEN_RESPONSE" | jq -r '.message')" != "Forbidden" ]]; then
  echo "Technician-token test failed: message should be Forbidden"
  exit 1
fi

echo "Technician-token test SUCCESS"
echo

echo "10. Test basic search by deviceCategoryId, expected all 3 new technicians appear"
BASIC_RESPONSE=$(curl -s -X GET "$BASE_URL/api/customers/technicians?deviceCategoryId=$AIR_CONDITIONER_ID" \
  -H "Authorization: Bearer $CUSTOMER_TOKEN")

echo "$BASIC_RESPONSE" | jq

for id in "$TECH_A_ID" "$TECH_B_ID" "$TECH_C_ID"; do
  FOUND=$(echo "$BASIC_RESPONSE" | jq --arg id "$id" '[.data[] | select(.technicianProfileId == $id)] | length')
  if [[ "$FOUND" -lt 1 ]]; then
    echo "Basic search test failed: technician $id not found"
    exit 1
  fi
done

echo "Basic search test SUCCESS"
echo

echo "11. Test filter availabilityStatus=ONLINE, expected Alpha and Bravo appear, Charlie not appear"
ONLINE_RESPONSE=$(curl -s -X GET "$BASE_URL/api/customers/technicians?deviceCategoryId=$AIR_CONDITIONER_ID&availabilityStatus=ONLINE&sort=name" \
  -H "Authorization: Bearer $CUSTOMER_TOKEN")

echo "$ONLINE_RESPONSE" | jq

HAS_A_ONLINE=$(echo "$ONLINE_RESPONSE" | jq --arg id "$TECH_A_ID" '[.data[] | select(.technicianProfileId == $id)] | length')
HAS_B_ONLINE=$(echo "$ONLINE_RESPONSE" | jq --arg id "$TECH_B_ID" '[.data[] | select(.technicianProfileId == $id)] | length')
HAS_C_ONLINE=$(echo "$ONLINE_RESPONSE" | jq --arg id "$TECH_C_ID" '[.data[] | select(.technicianProfileId == $id)] | length')

if [[ "$HAS_A_ONLINE" -lt 1 || "$HAS_B_ONLINE" -lt 1 ]]; then
  echo "ONLINE filter test failed: Alpha or Bravo not found"
  exit 1
fi

if [[ "$HAS_C_ONLINE" -ne 0 ]]; then
  echo "ONLINE filter test failed: Charlie should not appear"
  exit 1
fi

echo "ONLINE filter test SUCCESS"
echo

echo "12. Test filter availabilityStatus=OFFLINE, expected Charlie appears"
OFFLINE_RESPONSE=$(curl -s -X GET "$BASE_URL/api/customers/technicians?deviceCategoryId=$AIR_CONDITIONER_ID&availabilityStatus=OFFLINE&sort=name" \
  -H "Authorization: Bearer $CUSTOMER_TOKEN")

echo "$OFFLINE_RESPONSE" | jq

HAS_C_OFFLINE=$(echo "$OFFLINE_RESPONSE" | jq --arg id "$TECH_C_ID" '[.data[] | select(.technicianProfileId == $id)] | length')

if [[ "$HAS_C_OFFLINE" -lt 1 ]]; then
  echo "OFFLINE filter test failed: Charlie not found"
  exit 1
fi

echo "OFFLINE filter test SUCCESS"
echo

echo "13. Test sort=rating, expected created technicians order: Charlie, Alpha, Bravo"
RATING_RESPONSE=$(curl -s -X GET "$BASE_URL/api/customers/technicians?deviceCategoryId=$AIR_CONDITIONER_ID&sort=rating" \
  -H "Authorization: Bearer $CUSTOMER_TOKEN")

echo "$RATING_RESPONSE" | jq

CREATED_RATING_ORDER=$(echo "$RATING_RESPONSE" | jq -r \
  --arg a "$TECH_A_ID" \
  --arg b "$TECH_B_ID" \
  --arg c "$TECH_C_ID" \
  '.data[]
   | select(.technicianProfileId == $a or .technicianProfileId == $b or .technicianProfileId == $c)
   | .technicianProfileId' | paste -sd "," -)

EXPECTED_RATING_ORDER="$TECH_C_ID,$TECH_A_ID,$TECH_B_ID"

echo "Created technician rating order  : $CREATED_RATING_ORDER"
echo "Expected technician rating order : $EXPECTED_RATING_ORDER"

if [[ "$CREATED_RATING_ORDER" != "$EXPECTED_RATING_ORDER" ]]; then
  echo "sort=rating test failed"
  exit 1
fi

echo "sort=rating test SUCCESS"
echo

echo "14. Test sort=totalJobs, expected created technicians order: Bravo, Alpha, Charlie"
TOTAL_JOBS_RESPONSE=$(curl -s -X GET "$BASE_URL/api/customers/technicians?deviceCategoryId=$AIR_CONDITIONER_ID&sort=totalJobs" \
  -H "Authorization: Bearer $CUSTOMER_TOKEN")

echo "$TOTAL_JOBS_RESPONSE" | jq

CREATED_TOTAL_JOBS_ORDER=$(echo "$TOTAL_JOBS_RESPONSE" | jq -r \
  --arg a "$TECH_A_ID" \
  --arg b "$TECH_B_ID" \
  --arg c "$TECH_C_ID" \
  '.data[]
   | select(.technicianProfileId == $a or .technicianProfileId == $b or .technicianProfileId == $c)
   | .technicianProfileId' | paste -sd "," -)

EXPECTED_TOTAL_JOBS_ORDER="$TECH_B_ID,$TECH_A_ID,$TECH_C_ID"

echo "Created technician totalJobs order  : $CREATED_TOTAL_JOBS_ORDER"
echo "Expected technician totalJobs order : $EXPECTED_TOTAL_JOBS_ORDER"

if [[ "$CREATED_TOTAL_JOBS_ORDER" != "$EXPECTED_TOTAL_JOBS_ORDER" ]]; then
  echo "sort=totalJobs test failed"
  exit 1
fi

echo "sort=totalJobs test SUCCESS"
echo

echo "15. Test sort=name, expected created technicians order: Alpha, Bravo, Charlie"
NAME_RESPONSE=$(curl -s -X GET "$BASE_URL/api/customers/technicians?deviceCategoryId=$AIR_CONDITIONER_ID&sort=name" \
  -H "Authorization: Bearer $CUSTOMER_TOKEN")

echo "$NAME_RESPONSE" | jq

CREATED_NAME_ORDER=$(echo "$NAME_RESPONSE" | jq -r \
  --arg a "$TECH_A_ID" \
  --arg b "$TECH_B_ID" \
  --arg c "$TECH_C_ID" \
  '.data[]
   | select(.technicianProfileId == $a or .technicianProfileId == $b or .technicianProfileId == $c)
   | .technicianProfileId' | paste -sd "," -)

EXPECTED_NAME_ORDER="$TECH_A_ID,$TECH_B_ID,$TECH_C_ID"

echo "Created technician name order  : $CREATED_NAME_ORDER"
echo "Expected technician name order : $EXPECTED_NAME_ORDER"

if [[ "$CREATED_NAME_ORDER" != "$EXPECTED_NAME_ORDER" ]]; then
  echo "sort=name test failed"
  exit 1
fi

echo "sort=name test SUCCESS"
echo

echo "16. Test invalid availabilityStatus, expected validation error"
INVALID_STATUS_RESPONSE=$(curl -s -X GET "$BASE_URL/api/customers/technicians?deviceCategoryId=$AIR_CONDITIONER_ID&availabilityStatus=NGACO" \
  -H "Authorization: Bearer $CUSTOMER_TOKEN")

echo "$INVALID_STATUS_RESPONSE" | jq

if [[ "$(echo "$INVALID_STATUS_RESPONSE" | jq -r '.success')" != "false" ]]; then
  echo "Invalid availabilityStatus test failed: success should be false"
  exit 1
fi

if [[ "$(echo "$INVALID_STATUS_RESPONSE" | jq -r '.message')" != "Invalid availability status. Allowed values: ONLINE, OFFLINE, BUSY, ON_LEAVE" ]]; then
  echo "Invalid availabilityStatus test failed: message mismatch"
  exit 1
fi

echo "Invalid availabilityStatus test SUCCESS"
echo

echo "17. Test invalid sort, expected validation error"
INVALID_SORT_RESPONSE=$(curl -s -X GET "$BASE_URL/api/customers/technicians?deviceCategoryId=$AIR_CONDITIONER_ID&sort=price" \
  -H "Authorization: Bearer $CUSTOMER_TOKEN")

echo "$INVALID_SORT_RESPONSE" | jq

if [[ "$(echo "$INVALID_SORT_RESPONSE" | jq -r '.success')" != "false" ]]; then
  echo "Invalid sort test failed: success should be false"
  exit 1
fi

if [[ "$(echo "$INVALID_SORT_RESPONSE" | jq -r '.message')" != "Invalid sort option. Allowed values: name, rating, totalJobs" ]]; then
  echo "Invalid sort test failed: message mismatch"
  exit 1
fi

echo "Invalid sort test SUCCESS"
echo

echo "18. Test invalid deviceCategoryId, expected Device category not found"
INVALID_CATEGORY_RESPONSE=$(curl -s -X GET "$BASE_URL/api/customers/technicians?deviceCategoryId=00000000-0000-0000-0000-000000000000&availabilityStatus=ONLINE&sort=rating" \
  -H "Authorization: Bearer $CUSTOMER_TOKEN")

echo "$INVALID_CATEGORY_RESPONSE" | jq

if [[ "$(echo "$INVALID_CATEGORY_RESPONSE" | jq -r '.success')" != "false" ]]; then
  echo "Invalid deviceCategoryId test failed: success should be false"
  exit 1
fi

if [[ "$(echo "$INVALID_CATEGORY_RESPONSE" | jq -r '.message')" != "Device category not found" ]]; then
  echo "Invalid deviceCategoryId test failed: message should be Device category not found"
  exit 1
fi

echo "Invalid deviceCategoryId test SUCCESS"
echo

echo "BE-26 SUCCESS: all tests passed"
