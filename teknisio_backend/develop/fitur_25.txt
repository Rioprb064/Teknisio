#!/usr/bin/env bash
set -euo pipefail

BASE_URL="http://localhost:8080"

SCRIPT_DIR="$(cd "$(dirname "${BASH_SOURCE[0]}")" && pwd)"

if [[ -f ".env" ]]; then
  ENV_FILE=".env"
elif [[ -f "$SCRIPT_DIR/../.env" ]]; then
  ENV_FILE="$SCRIPT_DIR/../.env"
else
  echo ".env not found. Run this script from backend root folder."
  exit 1
fi

for cmd in curl jq psql; do
  if ! command -v "$cmd" >/dev/null 2>&1; then
    echo "$cmd is not installed"
    exit 1
  fi
done

set -a
source "$ENV_FILE"
set +a

DB_HOST="${POSTGRES_HOST:-localhost}"
DB_PORT="${POSTGRES_PORT:-5433}"
DB_NAME="${POSTGRES_DB:-teknisio_db}"
DB_USER="${POSTGRES_USER:-teknisio_user}"
DB_PASSWORD="${POSTGRES_PASSWORD:-}"

if [[ -z "$DB_PASSWORD" ]]; then
  echo "POSTGRES_PASSWORD is empty. Check your .env file."
  exit 1
fi

TS=$(date +%s)

CUSTOMER_EMAIL="customerbe25${TS}@example.com"
CUSTOMER_PHONE="0812${TS: -8}"

TECHNICIAN_EMAIL="technicianbe25${TS}@example.com"
TECHNICIAN_PHONE="0813${TS: -8}"

PASSWORD="Password123"

echo "Using test data:"
echo "Customer email   : $CUSTOMER_EMAIL"
echo "Technician email : $TECHNICIAN_EMAIL"
echo "Database         : $DB_HOST:$DB_PORT/$DB_NAME"
echo

echo "1. Register customer"
CUSTOMER_REGISTER_RESPONSE=$(curl -s -X POST "$BASE_URL/api/auth/register/customer" \
  -H "Content-Type: application/json" \
  -d "{
    \"name\": \"Customer BE25 Test\",
    \"email\": \"$CUSTOMER_EMAIL\",
    \"phoneNumber\": \"$CUSTOMER_PHONE\",
    \"password\": \"$PASSWORD\",
    \"address\": \"Jl. Customer BE25 No. 1\"
  }")

echo "$CUSTOMER_REGISTER_RESPONSE" | jq

if [[ "$(echo "$CUSTOMER_REGISTER_RESPONSE" | jq -r '.success')" != "true" ]]; then
  echo "Customer registration failed"
  exit 1
fi

echo

echo "2. Register technician"
TECHNICIAN_REGISTER_RESPONSE=$(curl -s -X POST "$BASE_URL/api/auth/register/technician" \
  -H "Content-Type: application/json" \
  -d "{
    \"name\": \"Technician BE25 Test\",
    \"email\": \"$TECHNICIAN_EMAIL\",
    \"phoneNumber\": \"$TECHNICIAN_PHONE\",
    \"password\": \"$PASSWORD\",
    \"address\": \"Jl. Technician BE25 No. 2\",
    \"description\": \"BE25 technician detail test for Air Conditioner and Refrigerator\"
  }")

echo "$TECHNICIAN_REGISTER_RESPONSE" | jq

if [[ "$(echo "$TECHNICIAN_REGISTER_RESPONSE" | jq -r '.success')" != "true" ]]; then
  echo "Technician registration failed"
  exit 1
fi

TECHNICIAN_PROFILE_ID=$(echo "$TECHNICIAN_REGISTER_RESPONSE" | jq -r '.data.technicianProfileId')

if [[ "$TECHNICIAN_PROFILE_ID" == "null" || -z "$TECHNICIAN_PROFILE_ID" ]]; then
  echo "technicianProfileId not found"
  exit 1
fi

echo "Technician profile ID: $TECHNICIAN_PROFILE_ID"
echo

echo "3. Login customer and auto-save token"
CUSTOMER_LOGIN_RESPONSE=$(curl -s -X POST "$BASE_URL/api/auth/login" \
  -H "Content-Type: application/json" \
  -d "{
    \"email\": \"$CUSTOMER_EMAIL\",
    \"password\": \"$PASSWORD\"
  }")

CUSTOMER_TOKEN=$(echo "$CUSTOMER_LOGIN_RESPONSE" | jq -r '.data.accessToken')
echo "$CUSTOMER_LOGIN_RESPONSE" | jq '.data.accessToken = "<hidden>" | .data.refreshToken = "<hidden>"'

if [[ "$CUSTOMER_TOKEN" == "null" || -z "$CUSTOMER_TOKEN" ]]; then
  echo "Customer token not found"
  exit 1
fi

echo

echo "4. Login technician and auto-save token"
TECHNICIAN_LOGIN_RESPONSE=$(curl -s -X POST "$BASE_URL/api/auth/login" \
  -H "Content-Type: application/json" \
  -d "{
    \"email\": \"$TECHNICIAN_EMAIL\",
    \"password\": \"$PASSWORD\"
  }")

TECHNICIAN_TOKEN=$(echo "$TECHNICIAN_LOGIN_RESPONSE" | jq -r '.data.accessToken')
echo "$TECHNICIAN_LOGIN_RESPONSE" | jq '.data.accessToken = "<hidden>" | .data.refreshToken = "<hidden>"'

if [[ "$TECHNICIAN_TOKEN" == "null" || -z "$TECHNICIAN_TOKEN" ]]; then
  echo "Technician token not found"
  exit 1
fi

echo

echo "5. Get device categories"
DEVICE_CATEGORIES_RESPONSE=$(curl -s -X GET "$BASE_URL/api/device-categories")
echo "$DEVICE_CATEGORIES_RESPONSE" | jq

AIR_CONDITIONER_ID=$(echo "$DEVICE_CATEGORIES_RESPONSE" | jq -r '.data[] | select(.name == "Air Conditioner") | .deviceCategoryId')
REFRIGERATOR_ID=$(echo "$DEVICE_CATEGORIES_RESPONSE" | jq -r '.data[] | select(.name == "Refrigerator") | .deviceCategoryId')

if [[ "$AIR_CONDITIONER_ID" == "null" || -z "$AIR_CONDITIONER_ID" ]]; then
  echo "Air Conditioner category not found"
  exit 1
fi

if [[ "$REFRIGERATOR_ID" == "null" || -z "$REFRIGERATOR_ID" ]]; then
  echo "Refrigerator category not found"
  exit 1
fi

echo "Air Conditioner ID: $AIR_CONDITIONER_ID"
echo "Refrigerator ID   : $REFRIGERATOR_ID"
echo

echo "6. Auto insert technician skills into real backend database"
PGPASSWORD="$DB_PASSWORD" psql \
  -h "$DB_HOST" \
  -p "$DB_PORT" \
  -U "$DB_USER" \
  -d "$DB_NAME" \
  -v ON_ERROR_STOP=1 \
  -c "
INSERT INTO teknisi_kategori_layanan (
  id_teknisi_profile,
  id_kategori,
  aktif
)
VALUES
  ('$TECHNICIAN_PROFILE_ID', '$AIR_CONDITIONER_ID', true),
  ('$TECHNICIAN_PROFILE_ID', '$REFRIGERATOR_ID', true)
ON CONFLICT (id_teknisi_profile, id_kategori)
DO UPDATE SET
  aktif = true,
  updated_at = CURRENT_TIMESTAMP;
"

echo

echo "7. Verify inserted technician skills"
PGPASSWORD="$DB_PASSWORD" psql \
  -h "$DB_HOST" \
  -p "$DB_PORT" \
  -U "$DB_USER" \
  -d "$DB_NAME" \
  -c "
SELECT
  u.nama AS technician_name,
  k.nama_kategori AS device_category,
  tkl.aktif
FROM teknisi_kategori_layanan tkl
JOIN teknisi_profile tp
  ON tp.id_teknisi_profile = tkl.id_teknisi_profile
JOIN users u
  ON u.id_user = tp.id_user
JOIN kategori_layanan k
  ON k.id_kategori = tkl.id_kategori
WHERE tkl.id_teknisi_profile = '$TECHNICIAN_PROFILE_ID'
ORDER BY k.nama_kategori;
"

echo

echo "8. Prerequisite test BE-24: customer can search technicians by Air Conditioner"
BE24_RESPONSE=$(curl -s -X GET "$BASE_URL/api/customers/technicians?deviceCategoryId=$AIR_CONDITIONER_ID" \
  -H "Authorization: Bearer $CUSTOMER_TOKEN")

echo "$BE24_RESPONSE" | jq

BE24_CURRENT_TECHNICIAN_COUNT=$(echo "$BE24_RESPONSE" | jq --arg id "$TECHNICIAN_PROFILE_ID" '[.data[] | select(.technicianProfileId == $id)] | length')

if [[ "$BE24_CURRENT_TECHNICIAN_COUNT" -lt 1 ]]; then
  echo "BE-24 prerequisite failed: newly registered technician not found in search result"
  exit 1
fi

echo "BE-24 prerequisite SUCCESS: newly registered technician appears"
echo

echo "9. Test BE-25 without token, expected Unauthorized"
BE25_NO_TOKEN_RESPONSE=$(curl -s -X GET "$BASE_URL/api/customers/technicians/$TECHNICIAN_PROFILE_ID")
echo "$BE25_NO_TOKEN_RESPONSE" | jq

if [[ "$(echo "$BE25_NO_TOKEN_RESPONSE" | jq -r '.success')" != "false" ]]; then
  echo "BE-25 no-token test failed: success should be false"
  exit 1
fi

if [[ "$(echo "$BE25_NO_TOKEN_RESPONSE" | jq -r '.message')" != "Unauthorized" ]]; then
  echo "BE-25 no-token test failed: message should be Unauthorized"
  exit 1
fi

echo "BE-25 no-token test SUCCESS"
echo

echo "10. Test BE-25 with technician token, expected Forbidden"
BE25_TECHNICIAN_TOKEN_RESPONSE=$(curl -s -X GET "$BASE_URL/api/customers/technicians/$TECHNICIAN_PROFILE_ID" \
  -H "Authorization: Bearer $TECHNICIAN_TOKEN")

echo "$BE25_TECHNICIAN_TOKEN_RESPONSE" | jq

if [[ "$(echo "$BE25_TECHNICIAN_TOKEN_RESPONSE" | jq -r '.success')" != "false" ]]; then
  echo "BE-25 technician-token test failed: success should be false"
  exit 1
fi

if [[ "$(echo "$BE25_TECHNICIAN_TOKEN_RESPONSE" | jq -r '.message')" != "Forbidden" ]]; then
  echo "BE-25 technician-token test failed: message should be Forbidden"
  exit 1
fi

echo "BE-25 technician-token test SUCCESS"
echo

echo "11. Test BE-25 with customer token, expected technician detail"
BE25_CUSTOMER_TOKEN_RESPONSE=$(curl -s -X GET "$BASE_URL/api/customers/technicians/$TECHNICIAN_PROFILE_ID" \
  -H "Authorization: Bearer $CUSTOMER_TOKEN")

echo "$BE25_CUSTOMER_TOKEN_RESPONSE" | jq

if [[ "$(echo "$BE25_CUSTOMER_TOKEN_RESPONSE" | jq -r '.success')" != "true" ]]; then
  echo "BE-25 customer-token test failed: success should be true"
  exit 1
fi

if [[ "$(echo "$BE25_CUSTOMER_TOKEN_RESPONSE" | jq -r '.message')" != "Technician retrieved successfully" ]]; then
  echo "BE-25 customer-token test failed: message should be Technician retrieved successfully"
  exit 1
fi

DETAIL_TECHNICIAN_ID=$(echo "$BE25_CUSTOMER_TOKEN_RESPONSE" | jq -r '.data.technicianProfileId')
DETAIL_NAME=$(echo "$BE25_CUSTOMER_TOKEN_RESPONSE" | jq -r '.data.name')
DETAIL_DESCRIPTION=$(echo "$BE25_CUSTOMER_TOKEN_RESPONSE" | jq -r '.data.description')
DETAIL_SKILL_COUNT=$(echo "$BE25_CUSTOMER_TOKEN_RESPONSE" | jq '.data.supportedDeviceCategories | length')

HAS_AIR_CONDITIONER=$(echo "$BE25_CUSTOMER_TOKEN_RESPONSE" | jq --arg id "$AIR_CONDITIONER_ID" '[.data.supportedDeviceCategories[] | select(.deviceCategoryId == $id)] | length')
HAS_REFRIGERATOR=$(echo "$BE25_CUSTOMER_TOKEN_RESPONSE" | jq --arg id "$REFRIGERATOR_ID" '[.data.supportedDeviceCategories[] | select(.deviceCategoryId == $id)] | length')

if [[ "$DETAIL_TECHNICIAN_ID" != "$TECHNICIAN_PROFILE_ID" ]]; then
  echo "BE-25 customer-token test failed: technicianProfileId mismatch"
  exit 1
fi

if [[ "$DETAIL_NAME" != "Technician BE25 Test" ]]; then
  echo "BE-25 customer-token test failed: technician name mismatch"
  exit 1
fi

if [[ "$DETAIL_DESCRIPTION" != "BE25 technician detail test for Air Conditioner and Refrigerator" ]]; then
  echo "BE-25 customer-token test failed: description mismatch"
  exit 1
fi

if [[ "$DETAIL_SKILL_COUNT" -lt 2 ]]; then
  echo "BE-25 customer-token test failed: supportedDeviceCategories should contain at least 2 items"
  exit 1
fi

if [[ "$HAS_AIR_CONDITIONER" -lt 1 ]]; then
  echo "BE-25 customer-token test failed: Air Conditioner skill not found"
  exit 1
fi

if [[ "$HAS_REFRIGERATOR" -lt 1 ]]; then
  echo "BE-25 customer-token test failed: Refrigerator skill not found"
  exit 1
fi

echo "BE-25 customer-token detail test SUCCESS"
echo

echo "12. Test BE-25 invalid technician ID, expected Technician not found"
BE25_INVALID_RESPONSE=$(curl -s -X GET "$BASE_URL/api/customers/technicians/00000000-0000-0000-0000-000000000000" \
  -H "Authorization: Bearer $CUSTOMER_TOKEN")

echo "$BE25_INVALID_RESPONSE" | jq

if [[ "$(echo "$BE25_INVALID_RESPONSE" | jq -r '.success')" != "false" ]]; then
  echo "BE-25 invalid-id test failed: success should be false"
  exit 1
fi

if [[ "$(echo "$BE25_INVALID_RESPONSE" | jq -r '.message')" != "Technician not found" ]]; then
  echo "BE-25 invalid-id test failed: message should be Technician not found"
  exit 1
fi

echo "BE-25 invalid-id test SUCCESS"
echo

echo "BE-25 SUCCESS: all tests passed"
