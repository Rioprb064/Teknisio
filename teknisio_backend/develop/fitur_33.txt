#!/usr/bin/env bash
set -euo pipefail

BASE_URL="http://localhost:8080"

for cmd in curl jq; do
  if ! command -v "$cmd" >/dev/null 2>&1; then
    echo "$cmd is not installed"
    exit 1
  fi
done

TS=$(date +%s)
PASSWORD="Password123"

CUSTOMER_EMAIL="customerbe33${TS}@example.com"
CUSTOMER_PHONE="0812${TS: -8}"

TECH_EMAIL="techbe33${TS}@example.com"
TECH_PHONE="0821${TS: -8}"

echo "Using test data:"
echo "Customer email   : $CUSTOMER_EMAIL"
echo "Technician email : $TECH_EMAIL"
echo

echo "1. Register customer"
CUSTOMER_REGISTER_RESPONSE=$(curl -s -X POST "$BASE_URL/api/auth/register/customer" \
  -H "Content-Type: application/json" \
  -d "{
    \"name\": \"Customer BE33 Test\",
    \"email\": \"$CUSTOMER_EMAIL\",
    \"phoneNumber\": \"$CUSTOMER_PHONE\",
    \"password\": \"$PASSWORD\",
    \"address\": \"Jl. Customer BE33 No. 1\"
  }")

echo "$CUSTOMER_REGISTER_RESPONSE" | jq

if [[ "$(echo "$CUSTOMER_REGISTER_RESPONSE" | jq -r '.success')" != "true" ]]; then
  echo "Customer registration failed"
  exit 1
fi

echo

echo "2. Register technician"
TECH_REGISTER_RESPONSE=$(curl -s -X POST "$BASE_URL/api/auth/register/technician" \
  -H "Content-Type: application/json" \
  -d "{
    \"name\": \"Technician BE33 Test\",
    \"email\": \"$TECH_EMAIL\",
    \"phoneNumber\": \"$TECH_PHONE\",
    \"password\": \"$PASSWORD\",
    \"address\": \"Jl. Technician BE33 No. 1\",
    \"description\": \"Technician for BE33 status testing\"
  }")

echo "$TECH_REGISTER_RESPONSE" | jq

if [[ "$(echo "$TECH_REGISTER_RESPONSE" | jq -r '.success')" != "true" ]]; then
  echo "Technician registration failed"
  exit 1
fi

TECHNICIAN_PROFILE_ID=$(echo "$TECH_REGISTER_RESPONSE" | jq -r '.data.technicianProfileId')
echo "Technician profile ID: $TECHNICIAN_PROFILE_ID"
echo

echo "3. Login customer"
CUSTOMER_LOGIN_RESPONSE=$(curl -s -X POST "$BASE_URL/api/auth/login" \
  -H "Content-Type: application/json" \
  -d "{
    \"email\": \"$CUSTOMER_EMAIL\",
    \"password\": \"$PASSWORD\"
  }")

CUSTOMER_TOKEN=$(echo "$CUSTOMER_LOGIN_RESPONSE" | jq -r '.data.accessToken')
echo "$CUSTOMER_LOGIN_RESPONSE" | jq '.data.accessToken = "<hidden>" | .data.refreshToken = "<hidden>"'

if [[ "$CUSTOMER_TOKEN" == "null" || -z "$CUSTOMER_TOKEN" ]]; then
  echo "Customer token not found"
  exit 1
fi

echo

echo "4. Login technician"
TECH_LOGIN_RESPONSE=$(curl -s -X POST "$BASE_URL/api/auth/login" \
  -H "Content-Type: application/json" \
  -d "{
    \"email\": \"$TECH_EMAIL\",
    \"password\": \"$PASSWORD\"
  }")

TECH_TOKEN=$(echo "$TECH_LOGIN_RESPONSE" | jq -r '.data.accessToken')
echo "$TECH_LOGIN_RESPONSE" | jq '.data.accessToken = "<hidden>" | .data.refreshToken = "<hidden>"'

if [[ "$TECH_TOKEN" == "null" || -z "$TECH_TOKEN" ]]; then
  echo "Technician token not found"
  exit 1
fi

echo

echo "5. Test without token, expected Unauthorized"
NO_TOKEN_RESPONSE=$(curl -s -X PATCH "$BASE_URL/api/technicians/me/status" \
  -H "Content-Type: application/json" \
  -d '{
    "availabilityStatus": "ONLINE"
  }')

echo "$NO_TOKEN_RESPONSE" | jq

if [[ "$(echo "$NO_TOKEN_RESPONSE" | jq -r '.success')" != "false" ]]; then
  echo "No-token test failed: success should be false"
  exit 1
fi

if [[ "$(echo "$NO_TOKEN_RESPONSE" | jq -r '.message')" != "Unauthorized" ]]; then
  echo "No-token test failed: message should be Unauthorized"
  exit 1
fi

echo "No-token test SUCCESS"
echo

echo "6. Test with customer token, expected Forbidden"
CUSTOMER_TOKEN_RESPONSE=$(curl -s -X PATCH "$BASE_URL/api/technicians/me/status" \
  -H "Authorization: Bearer $CUSTOMER_TOKEN" \
  -H "Content-Type: application/json" \
  -d '{
    "availabilityStatus": "ONLINE"
  }')

echo "$CUSTOMER_TOKEN_RESPONSE" | jq

if [[ "$(echo "$CUSTOMER_TOKEN_RESPONSE" | jq -r '.success')" != "false" ]]; then
  echo "Customer-token test failed: success should be false"
  exit 1
fi

if [[ "$(echo "$CUSTOMER_TOKEN_RESPONSE" | jq -r '.message')" != "Forbidden" ]]; then
  echo "Customer-token test failed: message should be Forbidden"
  exit 1
fi

echo "Customer-token test SUCCESS"
echo

echo "7. Test invalid status, expected validation error"
INVALID_STATUS_RESPONSE=$(curl -s -X PATCH "$BASE_URL/api/technicians/me/status" \
  -H "Authorization: Bearer $TECH_TOKEN" \
  -H "Content-Type: application/json" \
  -d '{
    "availabilityStatus": "NGACO"
  }')

echo "$INVALID_STATUS_RESPONSE" | jq

if [[ "$(echo "$INVALID_STATUS_RESPONSE" | jq -r '.success')" != "false" ]]; then
  echo "Invalid status test failed: success should be false"
  exit 1
fi

if [[ "$(echo "$INVALID_STATUS_RESPONSE" | jq -r '.message')" != "Invalid availability status. Allowed values: ONLINE, OFFLINE, BUSY, ON_LEAVE" ]]; then
  echo "Invalid status test failed: message mismatch"
  exit 1
fi

echo "Invalid status test SUCCESS"
echo

echo "8. Test blank status, expected validation error"
BLANK_STATUS_RESPONSE=$(curl -s -X PATCH "$BASE_URL/api/technicians/me/status" \
  -H "Authorization: Bearer $TECH_TOKEN" \
  -H "Content-Type: application/json" \
  -d '{
    "availabilityStatus": ""
  }')

echo "$BLANK_STATUS_RESPONSE" | jq

if [[ "$(echo "$BLANK_STATUS_RESPONSE" | jq -r '.success')" != "false" ]]; then
  echo "Blank status test failed: success should be false"
  exit 1
fi

echo "Blank status test SUCCESS"
echo

test_status_update() {
  local status="$1"

  echo "9.x Update status to $status"
  RESPONSE=$(curl -s -X PATCH "$BASE_URL/api/technicians/me/status" \
    -H "Authorization: Bearer $TECH_TOKEN" \
    -H "Content-Type: application/json" \
    -d "{
      \"availabilityStatus\": \"$status\"
    }")

  echo "$RESPONSE" | jq

  if [[ "$(echo "$RESPONSE" | jq -r '.success')" != "true" ]]; then
    echo "Update status $status failed: success should be true"
    exit 1
  fi

  if [[ "$(echo "$RESPONSE" | jq -r '.message')" != "Technician availability status updated successfully" ]]; then
    echo "Update status $status failed: message mismatch"
    exit 1
  fi

  if [[ "$(echo "$RESPONSE" | jq -r '.data.technicianProfileId')" != "$TECHNICIAN_PROFILE_ID" ]]; then
    echo "Update status $status failed: technicianProfileId mismatch"
    exit 1
  fi

  if [[ "$(echo "$RESPONSE" | jq -r '.data.availabilityStatus')" != "$status" ]]; then
    echo "Update status $status failed: availabilityStatus mismatch"
    exit 1
  fi

  PROFILE_RESPONSE=$(curl -s -X GET "$BASE_URL/api/auth/profile" \
    -H "Authorization: Bearer $TECH_TOKEN")

  echo "$PROFILE_RESPONSE" | jq

  PROFILE_STATUS=$(echo "$PROFILE_RESPONSE" | jq -r '.data.availabilityStatus')

  if [[ "$PROFILE_STATUS" != "$status" ]]; then
    echo "Profile status check failed: expected $status but got $PROFILE_STATUS"
    exit 1
  fi

  echo "Update status $status SUCCESS"
  echo
}

echo "9. Test all valid statuses"
test_status_update "ONLINE"
test_status_update "OFFLINE"
test_status_update "BUSY"
test_status_update "ON_LEAVE"

echo "BE-33 SUCCESS: all tests passed"
